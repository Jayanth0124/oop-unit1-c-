#include<iostream>
#include<math.h>
using namespace std;
class PerfectSquare
{
    int start,end;
public
    PerfectSquare(int s,int e):start(s),end(e)
    {
    }
    void getPerfectSquare()
    {
        int count=0;
        for(int i=start;i<=end;i++)
        {
            double sq=sqrt(i);
            if(sq==(int)sq)
            {
                int sum=0;
                int temp=i;
                while(temp>0)
                {
                    sum=sum+temp%10;
                    temp=temp/10;
                }
                if(sum<10)
                {
                    cout<<i<<" ";
                    count++;
                }

            }
        }
        if(count==0)
            cout<<"There is no perfect square in the given range";

    }
};
int main()
{
    int start,end;
    cout<<"Enter lower range:";
    cin>>start;
    cout<<"Enter upper range:";
    cin>>end;
    PerfectSquare obj(start,end);
    obj.getPerfectSquare();
    return 0;
}
